## Install

```bash
npm install --save my-fancy-button
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'my-fancy-button'
import 'my-fancy-button/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
